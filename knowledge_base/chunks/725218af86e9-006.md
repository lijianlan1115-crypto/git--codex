 def do_POST(self) -> None:
 path = urlparse(self.path).path
 if path != "/api/issues":
 json_response(self, {"error": "not_found"}, 404)
 return
 try:
 length = int(self.headers.get("Content-Length", "0"))
 content_type = self.headers.get("Content-Type", "")
 if content_type.startswith("multipart/form-data"):
 payload, files = parse_multipart(self, length)
 else:
 payload = json.loads(self.rfile.read(length).decode("utf-8"))
 files = []
 issue = make_issue(payload)
 if not issue.get("submitter") or not issue.get("skill_id"):
 json_response(self, {"error": "提交人、涉及 Skill 必填"}, 400)
 return
 issue["attachments"] = save_attachments(issue["id"], files)
 records = load_records()
 records.append(issue)
 append_issue_to_docx(issue)
 save_records(records)
 json_response(self, issue, 201)
 except Exception as exc:
 json_response(self, {"error": f"{type(exc).__name__}: {exc}"}, 500)

 def do_DELETE(self) -> None:
 path = urlparse(self.path).path
 prefix = "/api/issues/"
 if not path.startswith(prefix):
 json_response(self, {"error": "not_found"}, 404)
 return
 issue_id = Path(path[len(prefix) :]).name
 if not issue_id:
 json_response(self, {"error": "missing_issue_id"}, 400)
 return
 result = delete_issue(issue_id)
 if result["status"] == "not_found":
 json_response(self, result, 404)
 return
 json_response(self, result)
