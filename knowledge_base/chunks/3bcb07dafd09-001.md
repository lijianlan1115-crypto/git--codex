"""Feishu adapter for auto-triggering S14.

This file is for the Feishu entry service to import or copy. Feishu users do
not run Python manually: the already-running Feishu bot calls this adapter
after receiving a message event.
"""

from __future__ import annotations

import json
import subprocess
import sys
from datetime import datetime
from pathlib import Path
from typing import Any
from urllib.parse import parse_qsl, urlencode, urlsplit, urlunsplit

try:
 from .router import build_control_inputs, should_route_to_s14
 from .reply_formatter import FORMAT_ERROR_TEXT, format_agent_json_output, format_feishu_message
except ImportError:
 from router import build_control_inputs, should_route_to_s14
 from reply_formatter import FORMAT_ERROR_TEXT, format_agent_json_output, format_feishu_message

PACKAGE_ROOT = Path(__file__).resolve().parents[1]
LOCAL_RUNNER = PACKAGE_ROOT / "scripts/s14_local_report.py"

def _new_run_id() -> str:
 return datetime.now().strftime("%Y%m%d%H%M%S")

def _append_run_id(report_url: str, run_id: str) -> str:
 """Append/replace run_id so Feishu/browser cannot show a stale cached page."""

 if not report_url:
 return report_url
 parts = urlsplit(str(report_url))
 query = dict(parse_qsl(parts.query, keep_blank_values=True))
 query["run_id"] = run_id
 return urlunsplit((parts.scheme, parts.netloc, parts.path, urlencode(query), parts.fragment))

def _prepare_current_result(result: dict[str, Any]) -> dict[str, Any]:
 """Force the reply to be rebuilt from the current run result only."""

 prepared = dict(result or {})
 run_id = str(prepared.get("run_id") or _new_run_id())
 prepared["run_id"] = run_id
 if prepared.get("report_url"):
 prepared["report_url"] = _append_run_id(prepared["report_url"], run_id)
 # Do not trust any prebuilt message here. It may come from an older run or
 # from an Agent summary. The locked formatter must render the current data.
 prepared.pop("feishu_message", None)
 return prepared

def run_s14_local_table_mode(python_bin: str | None = None) -> dict[str, Any]:
 """Run current local-table S14 mode and return structured result.

 TODO(openclaw-runtime): replace this subprocess call with
 S14OperationDiagnosis(config).execute(inputs) after database/official
 OpenClaw runtime is connected.
 """

 cmd = [python_bin or sys.executable, "-B", str(LOCAL_RUNNER)]
 completed = subprocess.run(
 cmd,
 cwd=str(PACKAGE_ROOT.parent),
 check=True,
 capture_output=True,
 text=True,
 )
 return json.loads(completed.stdout)

def build_feishu_reply(result: dict[str, Any]) -> str:
 try:
 return format_feishu_message(_prepare_current_result(result))
 except Exception:
 return FORMAT_ERROR_TEXT

def build_feishu_reply_from_agent_output(agent_output: str) -> str:
 """Use when an Agent first produces JSON, then Python renders Feishu text."""

 return format_agent_json_output(agent_output)

def handle_feishu_text_message(text: str) -> str | None:
 """Return reply text if this message should trigger S14, otherwise None.

 Text trigger must run a fresh diagnosis. Never answer from previous Feishu
 messages, Agent memory, old JSON, or a prebuilt ``feishu_message``.
 """

 if not should_route_to_s14(text):
 return None
 _control_inputs = build_control_inputs(text)
 result = run_s14_local_table_mode()
 return build_feishu_reply(result)
