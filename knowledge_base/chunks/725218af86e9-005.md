 async function deleteIssue(issueId) {{
 if (!issueId) return;
 if (!confirm(`确认删除 ${{issueId}}？\\n\\n会从网页记录中删除，并移除对应截图文件；已经写入 Word 的历史内容会保留。`)) return;
 statusEl.className = '';
 statusEl.textContent = '正在删除 ' + issueId + '...';
 const res = await fetch('/api/issues/' + encodeURIComponent(issueId), {{ method: 'DELETE' }});
 const data = await res.json();
 if (!res.ok || data.status !== 'ok') {{
 statusEl.className = 'error';
 statusEl.textContent = data.error || '删除失败';
 return;
 }}
 statusEl.textContent = '已删除：' + issueId;
 await loadIssues();
 }}
