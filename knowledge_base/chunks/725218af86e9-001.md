ALL_FIELD_KEYS = sorted({key for template in TEMPLATES.values() for key, _label in template["fields"]})

def load_records() -> list[dict]:
 if not DATA_PATH.exists():
 return []
 try:
 data = json.loads(DATA_PATH.read_text(encoding="utf-8"))
 except json.JSONDecodeError:
 return []
 return data if isinstance(data, list) else []

def save_records(records: list[dict]) -> None:
 DATA_PATH.write_text(json.dumps(records, ensure_ascii=False, indent=2), encoding="utf-8")

def issue_number_from_id(issue_id: str | None) -> int:
 if not issue_id:
 return 0
 try:
 return int(str(issue_id).split("-")[-1])
 except ValueError:
 return 0

def next_issue_number() -> int:
 records = load_records()
 max_record_no = max((issue_number_from_id(item.get("id")) for item in records), default=0)
 counter_no = 0
 if COUNTER_PATH.exists():
 try:
 counter_no = int((json.loads(COUNTER_PATH.read_text(encoding="utf-8")) or {}).get("last_issue_no", 0))
 except (json.JSONDecodeError, TypeError, ValueError):
 counter_no = 0
 issue_no = max(max_record_no, counter_no) + 1
 COUNTER_PATH.write_text(json.dumps({"last_issue_no": issue_no}, ensure_ascii=False, indent=2), encoding="utf-8")
 return issue_no

def make_issue(payload: dict) -> dict:
 now = datetime.now()
 issue_no = next_issue_number()
 issue = {
 "id": f"ISSUE-{issue_no:03d}",
 "created_at": now.strftime("%Y-%m-%d %H:%M:%S"),
 "template_type": str(payload.get("template_type") or "C").strip() or "C",
 }
 if issue["template_type"] not in TEMPLATES:
 issue["template_type"] = "C"
 issue["template_name"] = TEMPLATES[issue["template_type"]]["name"]
 for key in ALL_FIELD_KEYS:
 issue[key] = str(payload.get(key, "")).strip()
 issue["title"] = str(payload.get("title", "")).strip()
 issue["attachments"] = payload.get("attachments") or []
 return issue

def safe_attachment_name(issue_id: str, original_name: str) -> str:
 suffix = Path(original_name or "").suffix.lower()
 if suffix not in IMAGE_EXTENSIONS:
 suffix = ".png"
 return f"{issue_id.lower()}-{uuid.uuid4().hex[:10]}{suffix}"

def save_attachments(issue_id: str, files: list[dict]) -> list[dict]:
 ATTACHMENTS_DIR.mkdir(parents=True, exist_ok=True)
 saved = []
 for file_item in files:
 filename = file_item.get("filename") or ""
 content = file_item.get("content") or b""
 if not content:
 continue
 name = safe_attachment_name(issue_id, filename)
 path = ATTACHMENTS_DIR / name
 path.write_bytes(content)
 saved.append(
 {
 "name": name,
 "original_name": filename or name,
 "url": f"/attachments/{name}",
 "path": str(path),
 }
 )
 return saved

def delete_issue(issue_id: str) -> dict:
 records = load_records()
 kept = []
 deleted = None
 for item in records:
 if item.get("id") == issue_id:
 deleted = item
 else:
 kept.append(item)
 if deleted is None:
 return {"status"
